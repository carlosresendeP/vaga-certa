geminiService.ts:import { GoogleGenAI, Type } from "@google/genai";
import { WorkoutPlan } from '../types';

export const generateFitnessPlan = async (userGoal: string): Promise<WorkoutPlan> => {
  const apiKey = process.env.API_KEY;
  if (!apiKey) {
    throw new Error("API Key is missing");
  }

  const ai = new GoogleGenAI({ apiKey });

  const prompt = `
    Você é um personal trainer de elite especializado em exercícios em casa (Home Workout).
    O usuário tem o seguinte objetivo: "${userGoal}".
    
    Crie um plano de treino completo e seguro.
    
    Regras:
    1. Se o usuário pediu X dias (ex: 30 dias), crie um cronograma para esse período. Se não especificou, assuma 30 dias.
    2. Crie "Rotinas" (A, B, C, Descanso, etc.) para evitar repetição excessiva de dados, e depois agende essas rotinas nos dias.
    3. Inclua benefícios claros e dicas de segurança cruciais para evitar lesões.
    4. Para cada exercício, gere um termo de busca otimizado para o YouTube (ex: "flexão correta tutorial").
    5. O idioma deve ser Português do Brasil.
    6. Seja motivador e realista.
  `;

  const response = await ai.models.generateContent({
    model: 'gemini-3-flash-preview',
    contents: prompt,
    config: {
      responseMimeType: "application/json",
      responseSchema: {
        type: Type.OBJECT,
        properties: {
          planName: { type: Type.STRING },
          overview: { type: Type.STRING, description: "Uma visão geral motivadora do plano." },
          routines: {
            type: Type.ARRAY,
            items: {
              type: Type.OBJECT,
              properties: {
                id: { type: Type.STRING },
                name: { type: Type.STRING },
                totalDurationMin: { type: Type.INTEGER },
                difficulty: { type: Type.STRING, enum: ["Iniciante", "Intermediário", "Avançado"] },
                exercises: {
                  type: Type.ARRAY,
                  items: {
                    type: Type.OBJECT,
                    properties: {
                      name: { type: Type.STRING },
                      description: { type: Type.STRING },
                      durationOrReps: { type: Type.STRING },
                      sets: { type: Type.INTEGER },
                      restBetweenSets: { type: Type.STRING },
                      benefits: { type: Type.STRING },
                      safetyTips: { type: Type.ARRAY, items: { type: Type.STRING } },
                      youtubeQuery: { type: Type.STRING }
                    },
                    required: ["name", "description", "durationOrReps", "sets", "safetyTips", "youtubeQuery", "benefits"]
                  }
                }
              },
              required: ["id", "name", "exercises", "difficulty", "totalDurationMin"]
            }
          },
          schedule: {
            type: Type.ARRAY,
            items: {
              type: Type.OBJECT,
              properties: {
                dayNumber: { type: Type.INTEGER },
                routineId: { type: Type.STRING },
                focus: { type: Type.STRING },
                isRestDay: { type: Type.BOOLEAN }
              },
              required: ["dayNumber", "routineId", "focus", "isRestDay"]
            }
          }
        },
        required: ["planName", "overview", "routines", "schedule"]
      }
    }
  });

  const jsonText = response.text;
  if (!jsonText) throw new Error("Falha ao gerar o plano.");
  
  return JSON.parse(jsonText) as WorkoutPlan;
};
